# Nested cross-validation

# define the number of folds for the outer and inner loops
n_outer_folds = 5
n_inner_folds = 5

# define lists to store the performance metrics for each fold
inner_scores_all = []
outer_scores_all = []
feature_importances_all = []

# perform nested cross-validation
outer_cv = KFold(n_splits=n_outer_folds, shuffle=True, random_state=seed)
for outer_fold, (outer_train_idx, outer_test_idx) in enumerate(outer_cv.split(X)):
    # split the data into training and testing sets for the outer fold
    X_outer_train, X_outer_test = X[outer_train_idx], X[outer_test_idx]
    y_outer_train, y_outer_test = y[outer_train_idx], y[outer_test_idx]
    
    # perform grid search for hyperparameter tuning using the inner loop
    inner_cv = KFold(n_splits=n_inner_folds, shuffle=True, random_state=seed)
    grid_search = GridSearchCV(estimator=RandomForestRegressor(random_state=seed), param_grid=param_grid, cv=inner_cv)
    inner_scores_fold = []
    feature_importances_fold = []
    for inner_fold, (inner_train_idx, inner_test_idx) in enumerate(inner_cv.split(X_outer_train)):
        # split the data into training and testing sets for the inner fold
        X_inner_train, X_inner_test = X_outer_train[inner_train_idx], X_outer_train[inner_test_idx]
        y_inner_train, y_inner_test = y_outer_train[inner_train_idx], y_outer_train[inner_test_idx]
        
        # fit the grid search object on the training set for the inner fold
        grid_search.fit(X_inner_train, y_inner_train)
        
        # evaluate the performance of the best model on the training set for the inner fold
        y_inner_train_pred = grid_search.best_estimator_.predict(X_inner_train)
        inner_train_scores = [metric(y_inner_train, y_inner_train_pred) for metric in metrics.values()]
        
        # evaluate the performance of the best model on the validation set for the inner fold
        y_inner_test_pred = grid_search.best_estimator_.predict(X_inner_test)
        inner_test_scores = [metric(y_inner_test, y_inner_test_pred) for metric in metrics.values()]
              
        # store the hyperparameters and performance metrics for this inner fold
        inner_scores_fold.append((grid_search.best_params_, inner_train_scores, inner_test_scores))
            
        # print the hyperparameters and performance metrics for this inner fold
        print('  Inner fold {} - Hyperparameters: {}'.format(inner_fold+1, grid_search.best_params_))
        print('    Training set:')
        for metric, score in zip(metrics.keys(), inner_train_scores):
            print('      {}: {:.3f}'.format(metric, score))
        print('    Validation set:')
        for metric, score in zip(metrics.keys(), inner_test_scores):
            print('      {}: {:.3f}'.format(metric, score))
            
    # evaluate the performance of the best model on the training set for the outer fold
    y_train_pred = grid_search.best_estimator_.predict(X_outer_train)
    train_scores = [metric(y_outer_train, y_train_pred) for metric in metrics.values()]
    
    # evaluate the performance of the best model on the test set for the outer fold
    y_test_pred = grid_search.best_estimator_.predict(X_outer_test)
    test_scores = [metric(y_outer_test, y_test_pred) for metric in metrics.values()]
    
    # store the performance metrics for this fold
    inner_scores_all.append(inner_scores_fold)
    outer_scores_all.append((grid_search.best_params_, train_scores, test_scores))

    # print the performance metrics for this fold
    print('Outer fold {} - Best hyperparameters: {}'.format(outer_fold+1, grid_search.best_params_))
    print('Training set:')
    for metric, score in zip(metrics.keys(), train_scores):
        print('  {}: {:.3f}'.format(metric, score))
    print('Testing set:')
    for metric, score in zip(metrics.keys(), test_scores):
        print('  {}: {:.3f}'.format(metric, score))
    
    # calculate the feature importance for this outer fold
    feature_importances = grid_search.best_estimator_.feature_importances_

    # get the feature numbers and feature names
    feature_names = ['ZST185Q01HA_1', 'ZAGE_1', 'ZESCS_1', 'ZICTRES_1', 'ZDISCLIMA_1', 'ZTEACHSUP_1', 'ZDIRINS_1', 
                     'ZPERFEED_1', 'ZEMOSUPS_1', 'ZSTIMREAD_1', 'ZADAPTIVITY_1', 'ZTEACHINT_1', 'ZJOYREAD_1', 'ZPERCOMP_1', 
                     'ZPERCOOP_1', 'ZATTLNACT_1', 'ZCOMPETE_1', 'ZWORKMAST_1', 'ZGFOFAIL_1', 'ZEUDMO_1', 'ZSWBP_1', 
                     'ZRESILIENCE_1', 'ZMASTGOAL_1', 'ZBELONG_1', 'ZBEINGBULLIED_1']


    # rank the variable importance for this outer fold
    feature_importances = np.mean([tree.feature_importances_ for tree in grid_search.best_estimator_.estimators_], axis=0)
    ranking = np.argsort(feature_importances)[::-1]
    print(f'Outer fold {outer_fold+1} - Feature importance ranking:')
    for rank, index in enumerate(ranking):
        feature = feature_names[index]
        importance = feature_importances[index]
        print(f'      Rank {rank+1}: {feature} - Importance: {importance:.3f}')
        
    # append the feature importance for this outer fold to the list
    feature_importances_fold.append(feature_importances)
